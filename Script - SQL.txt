create database if not exists Empresa;
use Empresa;

CREATE TABLE IF NOT EXISTS Alunos(
  nome			varchar(100),
  senha			varchar(30),
  telefone 		varchar(20),
  cpf			varchar(20) unique,
  email			varchar(50),
  sexo			varchar(15),
  curso			varchar(100),
  instituicao	varchar(100)  
);

CREATE TABLE IF NOT EXISTS Professores(
nome	varchar(100),
cpf		varchar(20) unique,
email	varchar(50),
sexo	varchar(15),
materia	varchar(100)
);
